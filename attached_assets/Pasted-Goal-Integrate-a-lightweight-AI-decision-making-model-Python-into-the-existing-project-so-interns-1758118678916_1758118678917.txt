# InternRecommender/server/model/recommender.py
# Simple, dependency-free Python recommender.
# Accepts a JSON object on stdin:
# { "education": "B.Tech - Computer Science", "interests": ["Information Technology"], "skills": ["Python","React"], "location": {"type":"offline"|"online","cities":["Chennai"]} }
# Outputs JSON list of recommendations to stdout.

import sys, json, math
from typing import List, Dict

def load_internships(path="server/data/internships.json"):
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)

def normalize(s: str):
    return s.strip().lower()

def skill_score(user_skills: List[str], intern_skills: List[str]) -> float:
    if not intern_skills:
        return 0.0
    u = [normalize(x) for x in user_skills]
    i = [normalize(x) for x in intern_skills]
    matched = 0
    for us in u:
        for is_ in i:
            # allow partial match (contains) for forgiving matching
            if us in is_ or is_ in us:
                matched += 1
                break
    return matched / max(1, len(i))
